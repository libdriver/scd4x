59 DRIVER_SCD4X_LINK_DELAY_MS(&gs_handle, scd4x_interface_delay_ms);
60 DRIVER_SCD4X_LINK_DEBUG_PRINT(&gs_handle, scd4x_interface_debug_print);
61
62 /* set chip type */
63 res = scd4x_set_type(&gs_handle, type);
64 if (res != 0)
65 {
66 scd4x_interface_debug_print("scd4x: set type failed.\n");
67
68 return 1;

70

75 scd4x_interface_debug_print("scd4x: init failed.\n");

79

82 if (res != 0)

88 }
